Getting Started

Quick Start

setup.ps1 PowerShell 
# Clone and setup
cd C:\workspace\Cthulu

# Create virtual environment
python -m venv venv
.\venv\Scripts\Activate.ps1

# Install dependencies
pip install -r requirements.txt
.env Environment 
# MT5 Credentials
MT5_LOGIN=your_account_number
MT5_PASSWORD=your_password
MT5_SERVER=YourBroker-MT5

# Risk Settings
RISK_PER_TRADE=0.02
MAX_DAILY_LOSS=0.05
run.sh Shell 
# Start autonomous trading
python __main__.py \
  --config config.example.json

# Or dry-run mode
python __main__.py \
  --config config.example.json \
  --dry-run
MT5 Setup Required 
# Enable algo trading in MT5:
